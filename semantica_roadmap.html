Phase 2: Query Engine (Weeks 3-4)

LangGraph Query Workflow

Agent Setup
from langgraph.graph import StateGraph workflow = StateGraph(QueryState) def supervisor_node(state: QueryState): """Analyze intent and route""" prompt = f"User query: {state['user_query']}" response = llm.invoke(prompt) state['intent'] = response.content return state def sql_agent_node(state: QueryState): """Generate SQL from natural language""" prompt = f""" Generate PostgreSQL query for: {state['user_query']} Available schema: {state['schema_context']} Rules: - Only SELECT queries allowed - Add LIMIT 1000 if not specified """ response = llm.invoke(prompt) state['generated_sql'] = response.content.strip() return state workflow.add_node("supervisor", supervisor_node) workflow.add_node("sql_agent", sql_agent_node) workflow.add_edge("supervisor", "sql_agent") app = workflow.compile()
